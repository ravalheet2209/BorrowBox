<script>
  document.addEventListener('DOMContentLoaded', function () {
    const pricePerDay = parseFloat("4999.00");
    const calendarEl = document.getElementById('bb-calendar');

    /* Format a date string (YYYY-MM-DD) for display */
    const fmt = s => {
      const d = new Date(s + 'T00:00:00');
      return d.toLocaleDateString('en-IN', { day: '2-digit', month: 'short', year: 'numeric' });
    };

    /* Collapse past-only rows so calendar doesn't show blank leading rows */
    function hideDisabledRows() {
      const rows = calendarEl.querySelectorAll('.fc-daygrid-body tbody tr');
      rows.forEach(row => {
        const cells = row.querySelectorAll('td.fc-day');
        const allDisabled = cells.length > 0 && Array.from(cells).every(
          td => td.classList.contains('fc-day-disabled') || td.classList.contains('fc-day-past')
        );
        row.style.display = allDisabled ? 'none' : '';
      });
    }

    /* Update booking summary UI */
    function showSummary(startStr, endStr) {
      /* endStr from FullCalendar is exclusive (next day), so show inclusive end */
      const endDisplay = new Date(endStr + 'T00:00:00');
      endDisplay.setDate(endDisplay.getDate() - 1);
      const endDisplayStr = endDisplay.toISOString().split('T')[0];

      const days = Math.round((new Date(endStr) - new Date(startStr)) / 86400000);

      document.getElementById('start_date').value = startStr;
      document.getElementById('end_date').value = endStr;
      document.getElementById('total_days').textContent = days;
      document.getElementById('total_price').textContent = (days * pricePerDay).toFixed(2);
      document.getElementById('display-start').textContent = fmt(startStr);
      document.getElementById('display-end').textContent = fmt(endDisplayStr);

      const summary = document.getElementById('booking-summary');
      summary.style.display = 'flex';
      summary.style.flexDirection = 'column';
    }

    let calInstance = null;

    function buildCalendar(bookings) {
      const today = new Date().toISOString().split('T')[0];

      calInstance = new FullCalendar.Calendar(calendarEl, {
        initialView: 'dayGridMonth',
        initialDate: today,
        selectable: true,
        selectMirror: true,
        unselectAuto: false,
        headerToolbar: { left: 'prev', center: 'title', right: 'next' },
        validRange: { start: today },
        height: 'auto',
        fixedWeekCount: false,
        showNonCurrentDates: false,

        events: bookings.map(r => ({
          start: r.start, end: r.end, display: 'background', color: '#ef4444'
        })),

        datesSet: function () {
          setTimeout(hideDisabledRows, 0);
        },

        /* Allow selection only if it doesn't overlap booked ranges.
           FullCalendar sends endStr as exclusive (day after last selected). */
        selectAllow: function (info) {
          const selStart = new Date(info.startStr + 'T00:00:00');
          /* endStr is exclusive — subtract 1 for the real last selected day */
          const selEnd = new Date(info.endStr + 'T00:00:00');
          selEnd.setDate(selEnd.getDate() - 1);

          for (let r of bookings) {
            const bs = new Date(r.start + 'T00:00:00');
            /* booking end from backend is exclusive (FullCalendar bg-event style) */
            const be = new Date(r.end + 'T00:00:00');
            be.setDate(be.getDate() - 1); /* inclusive last booked day */

            /* overlap: selStart <= be AND selEnd >= bs */
            if (selStart <= be && selEnd >= bs) return false;
          }
          return true;
        },

        select: function (info) {
          /* Single-day click: startStr = that day, endStr = next day → days = 1 ✓ */
          const days = Math.round((new Date(info.endStr) - new Date(info.startStr)) / 86400000);
          if (days < 1) { calInstance.unselect(); return; }
          showSummary(info.startStr, info.endStr);
        }
      });

      calInstance.render();

      /* Fix width=0 issue from CSS flex reorder on mobile */
      setTimeout(() => calInstance.updateSize(), 60);
      setTimeout(() => calInstance.updateSize(), 400);

      /* Stay correctly sized when layout shifts */
      if (window.ResizeObserver) {
        const ro = new ResizeObserver(() => calInstance.updateSize());
        ro.observe(calendarEl.parentElement);
      }

      return calInstance;
    }

    fetch("/get-booked-dates/11/")
      .then(r => r.json())
      .then(bookings => buildCalendar(bookings))
      .catch(() => buildCalendar([]));
  });
